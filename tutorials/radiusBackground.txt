Crie um drawable xml file com o codigo abaixo.
set em radius o quanto quer de arredondamento


<?xml version="1.0" encoding="utf-8"?>
<shape xmlns:android="http://schemas.android.com/apk/res/android"
    android:shape="rectangle">

    <!-- Cor de fundo do RadioGroup -->
    <solid android:color="@color/your_background_color" />

    <!-- Arredondamento dos cantos -->
    <!-- Você pode definir um raio para todos os cantos -->
    <corners android:radius="12dp" />

    <!-- Ou definir raios específicos para cada canto, se necessário -->
    <!--
    <corners
        android:topLeftRadius="12dp"
        android:topRightRadius="12dp"
        android:bottomLeftRadius="12dp"
        android:bottomRightRadius="12dp" />
    -->

    <!-- Opcional: Adicionar uma borda -->
    <!--
    <stroke
        android:width="2dp"
        android:color="@color/your_border_color" />
    -->

    <!-- Opcional: Adicionar padding interno, se o conteúdo estiver muito próximo das bordas -->
    <!--
    <padding
        android:left="8dp"
        android:top="8dp"
        android:right="8dp"
        android:bottom="8dp" />
    -->

</shape>
